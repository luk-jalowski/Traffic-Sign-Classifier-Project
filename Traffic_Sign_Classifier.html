def translate_random(image):
    #Translates image in x and y axis
    row,col, _ = image.shape
    limit = 4
    dx,dy = np.random.randint(-limit, limit, 2)
    matrix = np.float32([[1,0,dx],
                         [0,1,dy]])
    translation = cv2.warpAffine(image, matrix, (col,row))
    translation = translation[:,:,np.newaxis]
    return translation

def scaling_random(image): 
    row,col, _ = image.shape
    # get random transform limits
    limits = np.random.randint(-5,5)
    # starting corners of the transform(corners of the image)
    begining_corners = np.float32([[0, 0],
                              [row, 0],
                              [0, col],
                              [row, col]])

    # end corners of the transform(after scaling)
    final_corners = np.float32([[limits,limits],
                            [row-limits,limits],
                            [limits,col-limits],
                            [row-limits,col-limits]])
    
    transform_matrix = cv2.getPerspectiveTransform(final_corners,begining_corners)
    scale = cv2.warpPerspective(image,transform_matrix,(row,col))
    scale = scale[:,:,np.newaxis]
    return scale

def ConvNetwork(x):    
    # Arguments used for tf.truncated_normal, randomly defines variables for the weights and biases for each layer
    mu = 0
    sigma = 0.1

    conv1_weights = tf.Variable(tf.truncated_normal((5,5,1,48), mean=mu, stddev=sigma))
    conv1_bias = tf.Variable(tf.zeros(48))
    conv1 = tf.nn.conv2d(x, conv1_weights, strides = [1,1,1,1], padding='VALID') + conv1_bias
    
    conv1 = tf.nn.relu(conv1)
    
    conv1 = tf.nn.max_pool(conv1, ksize = [1,2,2,1], strides=[1,2,2,1], padding='VALID')

    conv2_weights = tf.Variable(tf.truncated_normal((5,5,48,96), mean = mu, stddev = sigma))
    conv2_bias = tf.Variable(tf.zeros(96))
    conv2 = tf.nn.conv2d(conv1, conv2_weights, strides=[1,1,1,1], padding='VALID') + conv2_bias
    
    conv2 = tf.nn.relu(conv2)
    
    conv2 = tf.nn.max_pool(conv2, ksize = [1,2,2,1], strides=[1,2,2,1], padding='VALID')
    
    conv3_weights = tf.Variable(tf.truncated_normal((3,3,96,172), mean = mu, stddev = sigma))
    conv3_bias = tf.Variable(tf.zeros(172))
    conv3 = tf.nn.conv2d(conv2, conv3_weights, strides=[1,1,1,1], padding='VALID') + conv3_bias
    
    conv3 = tf.nn.max_pool(conv3, ksize = [1,2,2,1], strides=[1,1,1,1], padding='VALID')
    
    conv3flatten = flatten(conv3) 
    
    fc1 = tf.nn.relu(conv3flatten)
    
    fc2_weights = tf.Variable(tf.truncated_normal(shape=(688,84), mean=mu, stddev=sigma))
    fc2_bias = tf.Variable(tf.zeros(84))
    fc2 = tf.matmul(fc1, fc2_weights) + fc2_bias
    
    fc2 = tf.nn.relu(fc2)
    
    fc2 = tf.nn.dropout(fc2, keep_prob)
    

    fc3_weights = tf.Variable(tf.truncated_normal(shape=(84,n_classes), mean = mu, stddev=sigma))
    fc3_bias = tf.Variable(tf.zeros(n_classes))
    logits = tf.matmul(fc2, fc3_weights) + fc3_bias
    
    return logits

def evaluate(X_data, y_data):
    num_examples = len(X_data)
    total_accuracy = 0
    sess = tf.get_default_session()
    for offset in range(0, num_examples, batch_size):
        batch_x, batch_y = X_data[offset:offset+batch_size], y_data[offset:offset+batch_size]
        accuracy = sess.run(accuracy_operation, feed_dict={x: batch_x, y: batch_y, keep_prob: 1.0})
        total_accuracy += (accuracy * len(batch_x))
    return total_accuracy / num_examples

I've decided to use trial and error approach when looking for the best architecture for my network. I've started with LeNet architecture. Then I decided to add one more 3x3 convolutional layer and dropout layer which improved accuracy a lot. Since accuracy was above 93% I've decided to stick with that network.

Final network architecture:

  1. 5x5 convolution (32x32x1 in, 28x28x48 out)
  2. ReLU
  3. 2x2 max pool (28x28x48 in, 14x14x48 out)
  4. 5x5 convolution (14x14x48 in, 10x10x96 out)
  5. ReLU
  6. 2x2 max pool (10x10x96 in, 5x5x96 out)
  7. 3x3 convolution (5x5x96 in, 3x3x172 out)
  8. 2x2 max pool (3x3x172 in, 2x2x172 out)
  9. Flatten
  10. ReLu
  11. Fully connected layer (688 in, 84 out)
  12. ReLu
  13. Dropout layer
  14. Fully connected layer (84 in, number of classes(43))
